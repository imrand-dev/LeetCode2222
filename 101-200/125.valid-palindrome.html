125. Valid Palindrome¶

Content¶

Given a string s, determine if it is a palindrome, considering only alphanumeric characters and ignoring cases.

 

Example 1:

Input: s = "A man, a plan, a canal: Panama"
Output: true
Explanation: "amanaplanacanalpanama" is a palindrome.

Example 2:

Input: s = "race a car"
Output: false
Explanation: "raceacar" is not a palindrome.

 

Constraints:

  • 1 <= s.length <= 2 * 105
  • s consists only of printable ASCII characters.

from typing import Tuple

class Solution:
    def isPalindrome(self, s: str) -> bool:
        left = 0
        right = len(s) - 1
        while left < right:
            is_left_valid, left_char = self.isValidChar(s[left])
            if not is_left_valid:
                left += 1
                continue
            
            is_right_valid, right_char = self.isValidChar(s[right])
            if not is_right_valid:
                right -= 1
                continue
            
            if left_char != right_char:
                return False
            left += 1
            right -= 1
            
        return True
            
    def isValidChar(self, c: str) -> Tuple[bool, str]:
        if (c >= '0' and c <= '9') or (c >= 'a' and c <= 'z'):
            return [True, c]
        if c >= 'A' and c <= 'Z':
            return [True, c.lower()]
        return [False, c]
